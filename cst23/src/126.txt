#include <iostream>
#include <algorithm>

using namespace std;

int 			a[1100];
int 			b[1100];

int main()
{
	int 	N;
	cin >> N;

	while (N--)
	{
		int 	n;
		cin >> n;
		for (int i = 0; i < n; i++)
			cin >> a[i];
		for (int i = 0; i < n; i++)
			cin >> b[i];

		sort(a, a + n);
		sort(b, b + n);
		reverse(a, a + n);
		int sum = 0;
		for (int i = 0; i < n; i++)
			sum += a[i] * b[i];
		cout << sum << '
'; 
	}

	return 0;
}
