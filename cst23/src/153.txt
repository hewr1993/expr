using namespace std;
int xi[107]={0};
int n;
int main()
{
	cin>>n;
	for(int i=1;i<n+2;i++)
	  cin>>xi[i];
	if(xi[1]==1)
	{
	}
	else if(xi[1]==-1)
	{
		cout<<"-";
	}
	else
	  cout<<xi[1];
	cout<<"x^"<<n;
	for(int i=2;i<n+2;i++)
	{
		if(xi[i]!=0)
		{
			if(xi[i]>0)
			  cout<<"+";
			else
			  cout<<"-";
			int a=abs(xi[i]);
			if(a!=1||i==n+1)
			  cout<<a;
			if(i<n+1)
			{
				cout<<"x";
				if(i<n)
				  cout<<"^"<<n-i+1;
			}
		}
	}
	cout<<endl;
	//system("pause");
	return 0;
}
