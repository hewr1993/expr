#include <iostream>
using namespace std;
char num[4];
char tri(char x)
{
    return (x-'0'+5)%10+'0';
}
void exchange(int a, int b)
{
    char mid;
    mid=num[a];
    num[a]=num[b];
    num[b]=mid;
}
int main(){
    int k;
    cin>>num;
    for(k=0;k<4;k++)
        num[k]=tri(num[k]);
    exchange(0,3);
    exchange(1,2);
    int i;
    for(i=0;i<4;i++)
    {
        if((i==0)&&(num[i]=='0'))
            continue;
        else
            cout<<num[i];
   }
    return 0;
}
