//进制转换2
#include <iostream>
using namespace std;
struct Hex
{
    int left;
    char rest;
}; 
Hex place[100];
char tri(int y){
    if(y<10)
        return y+'0';
    else
        return y-10+'A';
    }   
void trihex(unsigned int x){ 
    int wei=-1,i;
    for(i=0;i<100;i++)
    {
        place[i].left=x%16;
        wei=wei+1;
        x=x/16;
        if(x==0)
            break;
    }
    int k;
    for(k=wei;k>=0;k--)
    {
        place[k].rest=tri(place[k].left);
        cout<<place[k].rest;
    }
}
int main(){
    unsigned int n;
    cin>>n;
    trihex(n);
    return 0;
}
